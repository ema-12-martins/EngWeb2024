db.contratos.aggregate([
  {
    $addFields: {
      precoContratualNumber: { $toDouble: { $replaceAll: { input: "$precoContratual", find: ",", replacement: "." } } }
    }
  },
  {
    $group: {
      _id: "$entidade_comunicante",
      montante_total: { $sum: "$precoContratualNumber" }
    }
  }
])
